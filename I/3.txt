==========HQ-SRV==========

useradd -u 2013 sshuser

echo "sshuser:P@ssw0rd" | chpasswd

usermod -aG wheel sshuser

echo "WHEEL_USERS ALL=(ALL:ALL) NOPASSWD: ALL" > /etc/sudoers.d/sshuser

su -l sshuser

sudo id

exit
